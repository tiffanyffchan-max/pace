// 核心修改：調整人數時，同步更新未填分數局數的人數限制 function adjustPlayers(d) { let n = playerCount + d; if (n < 2 || n > 12) return; if (d > 0) { if (playerNames.length < n) playerNames.push(`P${n}`); // 關鍵修改：當你增加人數，我會檢查所有「仲未入分」或者「最後一局」 // 等你可以喺加人後，即刻喺現有嘅局數入新加玩家嘅分 rounds.forEach(round => { let isBlank = round.scores.every(s => s === null || s === 0); if (isBlank || round === rounds[rounds.length - 1]) { round.count = n; } }); } else { playerNames.pop(); // 減人就維持返原本 logic } playerCount = n; renderTable(); saveToLocal(); } // 另外加多個「手動解鎖」功能：萬一仲係灰色，撳吓個格仔就會解鎖 function updateScore(r, p, v) { // 如果手動入分，自動將嗰局人數提升到足以容納呢位玩家 if (p >= rounds[r].count) { rounds[r].count = p + 1; } rounds[r].scores[p] = v === '' ? 0 : parseFloat(v); saveAndCalc(); }
